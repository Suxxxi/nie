library(rvest)
## Warning: 程辑包'rvest'是用R版本3.5.3 来建造的
## 载入需要的程辑包：xml2
## Warning: 程辑包'xml2'是用R版本3.5.3 来建造的
url<-'http://www.gov.cn/premier/2017-03/16/content_5177940.htm'
web<-read_html(url,encoding="utf-8")
position<-web %>% html_nodes("div.pages_content") %>% html_text()
library(jiebaR)
## Warning: 程辑包'jiebaR'是用R版本3.5.3 来建造的
## 载入需要的程辑包：jiebaRD
## Warning: 程辑包'jiebaRD'是用R版本3.5.3 来建造的
engine_s<-worker(stop_word = "stopwords.txt")
seg<-segment(position,engine_s)
f<-freq(seg)
f<-f[order(f[2],decreasing=TRUE),]
library(wordcloud2)
## Warning: 程辑包'wordcloud2'是用R版本3.5.3 来建造的
f<-f[1:150,]
wordcloud2(f, size = 0.8 ,shape='star')
